-- Drop previous table if exists
drop table if exists goal;

-- Create new goal table
create table goal (
  id bigint generated by default as identity primary key,
  user_id uuid references usermodel(id) on delete cascade,
  title text not null,                    -- matches frontend field "title"
  description text,                       -- matches frontend field "description"
  category text default 'General',        -- matches frontend field "category"
  target_amount numeric not null,         -- matches frontend field "target_amount"
  current_amount numeric default 0,       -- backend tracks progress
  status text default 'pending',          -- backend: 'pending' or 'completed'
  created_at timestamp default now(),
  updated_at timestamp default now(),
  
  -- Optional future enhancements
  priority text default 'medium',         -- optional, for UI sorting
  deadline date,                           -- optional, for time-bound goals
  notes text                               -- optional additional info
);
